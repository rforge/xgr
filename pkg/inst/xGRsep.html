function (data)
{
x <- gsub(":.*|chr", "", data)
x[x == "X"] <- 23
x[x == "Y"] <- 24
x <- as.numeric(x)
vec_sep <- cumsum(table(x))
vec_sep <- vec_sep[-length(vec_sep)]
invisible(vec_sep)
}
